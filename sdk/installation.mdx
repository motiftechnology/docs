## Installation

### npm

```bash
npm install @chatafrika/sdk
```

### yarn

```bash
yarn add @chatafrika/sdk
```

### pnpm

```bash
pnpm add @chatafrika/sdk
```

## Requirements

- **Node.js**: >= 18
- **Browser**: Modern browsers (Chrome, Firefox, Safari, Edge)
- **TypeScript**: Optional but recommended

## Basic Setup

### 1. Import the SDK

```typescript
import { ChatAfrika } from '@chatafrika/sdk';
```

### 2. Initialize the Client

```typescript
const client = new ChatAfrika({
  apiUrl: 'https://api.chaterafrika.com',
  wsUrl: 'wss://realtime.chaterafrika.com',
  sdkToken: 'your-sdk-token'
});
```

### 3. Connect

```typescript
await client.connect();
console.log('Connected to ChaterAfrika');
```

## Configuration Options

<ParamField name="apiUrl" type="string" required>
  Backend API URL
</ParamField>

<ParamField name="wsUrl" type="string" required>
  WebSocket URL for real-time features
</ParamField>

<ParamField name="sdkToken" type="string" required>
  SDK token (obtain from `/api/v1/auth/sdk-token`)
</ParamField>

<ParamField name="debug" type="boolean">
  Enable debug logging (default: false)
</ParamField>

<ParamField name="reconnectDelay" type="number">
  Initial reconnection delay in ms (default: 1000)
</ParamField>

<ParamField name="maxReconnectDelay" type="number">
  Maximum reconnection delay in ms (default: 30000)
</ParamField>

## Getting SDK Token

You need an SDK token to connect. Get it from your backend:

```typescript
async function getSDKToken(userId: string) {
  const response = await fetch('https://api.chaterafrika.com/api/v1/auth/sdk-token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      app_id: 'your-app-id',
      app_secret: 'your-app-secret',
      user_id: userId
    })
  });
  
  const { token } = await response.json();
  return token;
}

// Use the token
const token = await getSDKToken('customer-id');
const client = new ChatAfrika({
  apiUrl: 'https://api.chaterafrika.com',
  wsUrl: 'wss://realtime.chaterafrika.com',
  sdkToken: token
});
```

<Warning>
Never expose `app_secret` in client-side code. Get SDK tokens server-side.
</Warning>

## Complete Example

```typescript
import { ChatAfrika } from '@chatafrika/sdk';

async function initializeChat() {
  // Get SDK token (from your server)
  const token = await getSDKToken('customer-id');
  
  // Initialize SDK
  const client = new ChatAfrika({
    apiUrl: 'https://api.chaterafrika.com',
    wsUrl: 'wss://realtime.chaterafrika.com',
    sdkToken: token,
    debug: true // Enable in development
  });
  
  // Connect
  await client.connect();
  
  // Listen for connection state
  client.on('connection_state', (state) => {
    console.log('Connection state:', state);
  });
  
  return client;
}

// Initialize
const client = await initializeChat();
```

## TypeScript Support

The SDK includes full TypeScript definitions:

```typescript
import { ChatAfrika, Message, Conversation } from '@chatafrika/sdk';

const client: ChatAfrika = new ChatAfrika({ /* config */ });
const message: Message = await client.messages.send({ /* ... */ });
```

## Browser vs Node.js

### Browser

Works out of the box - uses native WebSocket API.

### Node.js

Install `ws` package:

```bash
npm install ws
```

```typescript
import WebSocket from 'ws';
import { ChatAfrika } from '@chatafrika/sdk';

const client = new ChatAfrika({
  apiUrl: 'https://api.chaterafrika.com',
  wsUrl: 'wss://realtime.chaterafrika.com',
  sdkToken: token,
  wsImplementation: WebSocket // For Node.js
});
```
